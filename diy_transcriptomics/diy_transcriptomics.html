#re-formatting and feature engg data for normalization and visualization----
library(edgeR)

gene_tpm<-data.frame(genes$abundance)
gene_count<-genes$counts
dge_list<-DGEList(gene_count)  #to creat dge list it need raw counts not the tpm 
#Counts per million 
dge_list_log<-cpm(dge_list,log=TRUE)
dge_list_log<-data.frame('gene_name'=rownames(dge_list_log),dge_list_log,row.names = NULL)
colnames(dge_list_log)<-c('gene_name',experiment_data$sample_name)
dge_list_log_pivot<-pivot_longer(data=dge_list_log,
                                 cols = CL08:HS05,
                                 names_to = 'sample_name',
                                 values_to = 'expression')
pl1<-ggplot(dge_list_log_pivot,aes(x=sample_name,y=expression))+
  geom_violin(aes(fill=sample_name),trim = F)+
  stat_summary(fun='median',size=0.1)+
  labs(x='sample name',y='log 2 expression',
       title='log2 count per million',
       subtitle = 'untrimmed - no normalization',
       caption = Sys.time())+
  coord_flip()
print(pl1)
